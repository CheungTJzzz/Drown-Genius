Deterministic Optimal Growth Model¶

  • Consider the following social planner's problem:
$$ \max_{c_t,k_{t+1}} \sum_{t=0}^{\infty} \beta^t \ln c_t, $$

s.t. $c_t+k_{t+1}=k_t^{\alpha}+(1-\delta)k_t$, and $k_0>0$ is given.

  • The recursive formulation is given by
$$ v(k_t)=\max_{c_t,k_{t+1}} [\ln c_t + \beta v(k_{t+1})], $$

s.t. $c_t+k_{t+1}=k_t^{\alpha}+(1-\delta)k_t$, and $k_0>0$ is given.

# Set model parameters

beta = 0.95
alpha = 1 / 3
delta = 0.1
kss = ((1/beta-(1-delta))/alpha)**(1/(alpha-1))
kmin = 0.5 * kss
kmax = 1.5 * kss
n = 500
kgrid = np.linspace(kmin, kmax, n)


def u(c):
    """utility function"""
    if c > 0:
        res = np.log(c)
    else:
        res = -np.inf
    return res


def budget(k, x):
    '''
    budget constraint
    k is state, x is control
    '''
    return k**alpha + (1-delta)*k - x


def V_current(k, k_next, V_next):
    """objective value function to be maximized"""
    c = budget(k, k_next)
    res = u(c) + beta * V_next
    return res


def V_max(k_index, k_start, V):
    '''choose best k_next to maximize V'''
    k = kgrid[k_index]
    k_bound = budget(k, 0)  # keep non-zero consumption
    V_max = -np.inf  # initialize maximum value

    # loop over possible k_next
    for j in range(k_start, n):
        k_next = kgrid[j]
        V_next = V[j]
        if k_next < k_bound:  # check feasibility
            V_new = V_current(k, k_next, V_next)
            if V_new > V_max:
                V_max = V_new
                g_k = k_next
                k_start = j
            else:
                break

    return V_max, g_k, k_start


def V_update(V):
    # initialize updated value function and policy function
    V_new = np.zeros(n)
    g_new = np.zeros(n)
    k_start = 0
    # loop over grid points
    for i in range(n):
        V_new[i], g_new[i], k_start = V_max(i, k_start, V)

    return V_new, g_new


def V_iteration(V_initial):
    V = V_initial
    error = np.inf
    count = 0
    max_iter = 1000
    tol = 1e-6

    while count < max_iter and error > tol:
        V_new, g_new = V_update(V)
        error = np.nanmax(np.abs(V_new - V))
        V = V_new
        count = count + 1

    if error > tol:
        print("Failed to converge!")
    else:
        print(f"\nConverged in {count} iterations.")
    return V_new, g_new
